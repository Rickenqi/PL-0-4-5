var a, b, c, r;
begin
        read(a);
        read(b);
        read(c);
        if a > b then
                begin
                        r := a;
                        a := b;
                        b := r;
                end;
        if b > c then
                begin
                        r := b;
                        b := c;
                        c := r;
                end;
        if a > b then
                begin
                        r := a;
                        a := b;
                        b := r;
                end;
        write(a);
        write(b);
        write(c);
end.
